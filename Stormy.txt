local function isAlive(plr)
    local char = plr and plr.Character
    if not char then return false end
    local hum = char:FindFirstChildOfClass("Humanoid")
    return hum and hum.Health > 0
end

local function getHead(plr)
    local char = plr and plr.Character
    if not char then return nil end
    return char:FindFirstChild("Head")
end

local function isTeammate(plr)
    if not CFG.TeamCheck then return false end
    if plr.Team and LocalPlayer.Team then
        return plr.Team == LocalPlayer.Team
    end
    return false
end

local function isVisible(targetPart)
    if not CFG.WallCheck then return true end
    if not targetPart then return false end
    local origin = Camera.CFrame.Position
    local dir    = (targetPart.Position - origin)
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Exclude
    params.FilterDescendantsInstances = {LocalPlayer.Character, Camera}
    local result = workspace:Raycast(origin, dir, params)
    if result then
        return result.Instance:IsDescendantOf(targetPart.Parent)
    end
    return true
end

local function screenCenter()
    local vp = Camera.ViewportSize
    return Vector2.new(vp.X / 2, vp.Y / 2)
end

local function getNearestHead()
    Camera = workspace.CurrentCamera
    local myChar = LocalPlayer.Character
    if not myChar then return nil, math.huge end
    local myHRP = myChar:FindFirstChild("HumanoidRootPart")
    if not myHRP then return nil, math.huge end

    local myPos      = myHRP.Position
    local bestTarget = nil
    local bestDist   = math.huge

    for _, plr in ipairs(Players:GetPlayers()) do
        if plr == LocalPlayer then continue end
        if not isAlive(plr) then continue end
        if isTeammate(plr) then continue end

        local head = getHead(plr)
        if not head then continue end
        if not head.Parent then continue end

        local dist = (head.Position - myPos).Magnitude

        if dist < bestDist then
            local canSee = true
            if CFG.WallCheck then
                canSee = isVisible(head)
            end
            if canSee then
                bestTarget = head
                bestDist   = dist
            end
        end
    end

    return bestTarget, bestDist
end

    local data = ESPObjects[plr]
    if not data then return end
    pcall(function() data.Highlight:Destroy() end)
    pcall(function() data.Billboard:Destroy() end)

local SilentAimTarget = nil

local oldIndex
pcall(function()
    local mt = getrawmetatable(game)
    oldIndex = mt.__index
    setreadonly(mt, false)
    mt.__index = newcclosure(function(self, key)
        if CFG.SilentAim and SilentAimTarget then
            if (key == "Hit" or key == "hit") and self == Mouse then
                return CFrame.new(oldIndex(self, key).Position, SilentAimTarget.Position)
            end
            if (key == "Target" or key == "target") and self == Mouse then
                return SilentAimTarget
            end
            if key == "UnitRay" and self == Mouse then
                local origin = Camera.CFrame.Position
                local dir    = (SilentAimTarget.Position - origin).Unit
                return Ray.new(origin, dir)
            end
        end
        return oldIndex(self, key)
    end)
    setreadonly(mt, true)
end)

local Connections = {}
local StoredCameraCF = nil
local CurrentTarget  = nil

Connections.Render = RunService.RenderStepped:Connect(function(dt)
    Camera = workspace.CurrentCamera

    if FOVCircle then
        FOVCircle.Visible  = CFG.ShowFOV
        FOVCircle.Position = screenCenter()
    end

    if CFG.Aimbot and isAlive(LocalPlayer) then
        local head, dist = getNearestHead()
        CurrentTarget = head

        if head and head.Parent then
            local camPos    = Camera.CFrame.Position
            local targetPos = head.Position
            Camera.CFrame   = CFrame.new(camPos, targetPos)
        end
    else
        CurrentTarget = nil
    end

    if LockDot then
        if CurrentTarget and CFG.Aimbot and CurrentTarget.Parent then
            local vec, onScr = Camera:WorldToViewportPoint(CurrentTarget.Position)
            LockDot.Visible  = onScr
            LockDot.Position = Vector2.new(vec.X, vec.Y)
        else
            LockDot.Visible = false
        end
    end

    if CFG.SilentAim and isAlive(LocalPlayer) then
        local head = getNearestHead()
        SilentAimTarget = head
    else
        SilentAimTarget = nil
    end

    if CFG.NoRecoil and isAlive(LocalPlayer) then
        if StoredCameraCF then
            local _, ry, _ = StoredCameraCF:ToEulerAnglesYXZ()
            local rx, _, rz = Camera.CFrame:ToEulerAnglesYXZ()
            Camera.CFrame = CFrame.new(Camera.CFrame.Position) * CFrame.Angles(rx, ry, rz)
        end
    end
    StoredCameraCF = Camera.CFrame

    for plr, data in pairs(ESPObjects) do
        if not isAlive(plr) then
            removeESP(plr)
        else
            local hrp = data.HRP
            local hum = data.Humanoid
            if not hrp or not hrp.Parent then removeESP(plr); continue end

            local espOn = CFG.ESP
            data.NameLabel.Visible   = espOn and CFG.ESPNames
            data.HealthLabel.Visible = espOn and CFG.ESPHealth
            data.DistLabel.Visible   = espOn and CFG.ESPDist

            if espOn then
                data.NameLabel.Text = plr.DisplayName
                if hum then
                    local hp  = math.floor(hum.Health)
                    local mhp = math.floor(hum.MaxHealth)
                    data.HealthLabel.Text = "HP: " .. hp .. "/" .. mhp
                    local pct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                    data.HealthLabel.TextColor3 = Color3.fromRGB(255*(1-pct), 255*pct, 50)
                end
                local d = (hrp.Position - Camera.CFrame.Position).Magnitude
                data.DistLabel.Text = "[" .. math.floor(d) .. " studs]"
            end

            local hl = data.Highlight
            if hl then
                hl.FillTransparency    = CFG.Chams and CFG.ChamsFill or 1
                hl.OutlineTransparency = CFG.Chams and CFG.ChamsOutline or 1
                hl.FillColor           = CFG.ChamsColor
            end

            if data.Tracer then
                if CFG.ESP and CFG.ESPTracers then
                    local vec, onScr = Camera:WorldToViewportPoint(hrp.Position)
                    data.Tracer.Visible = onScr
                    data.Tracer.From = Vector2.new(Camera.ViewportSize.X/2, Camera.ViewportSize.Y)
                    data.Tracer.To   = Vector2.new(vec.X, vec.Y)
                else
                    data.Tracer.Visible = false
                end
            end
        end
    end

    pcall(function()
        local char = LocalPlayer.Character
        if char then
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hum then
                if CFG.Speed then hum.WalkSpeed = CFG.SpeedVal end
                if CFG.JumpPower then hum.JumpPower = CFG.JumpVal; hum.UseJumpPower = true end
            end
        end
    end)

    if CFG.NoClip then
        pcall(function()
            for _, p in ipairs(LocalPlayer.Character:GetDescendants()) do
                if p:IsA("BasePart") then p.CanCollide = false end
            end
        end)
    end
end)

    Name     = "Rejoin Server",
    Callback = function()
        game:GetService("TeleportService"):TeleportToPlaceInstance(game.PlaceId, game.JobId)
    end,
})

MiscTab:CreateButton({
    Name     = "Destroy Script",
    Callback = function()
        for _, conn in pairs(Connections) do pcall(function() conn:Disconnect() end) end
        clearAllESP()
        if FOVCircle then FOVCircle:Remove() end
        if LockDot then LockDot:Remove() end
        if CrosshairGui then CrosshairGui:Destroy() end
        pcall(function()
            local mt = getrawmetatable(game)
            setreadonly(mt, false)
            mt.__index = oldIndex
            setreadonly(mt, true)
        end)
        Rayfield:Destroy()
    end,
})
